## 286. Walls and Gates (Medium) [link](https://leetcode.com/problems/walls-and-gates/)
[tag] BFS shortest path
- use multiple targets as starting points and add to queue

```python
class Solution:
    def wallsAndGates(self, rooms: List[List[int]]) -> None:
        """
        Do not return anything, modify rooms in-place instead.
        """
        ROW, COL = len(rooms), len(rooms[0])
        room_q = deque()
        visited = set()
        
        def add_room(x, y):
            if 0 <= x < ROW and 0 <= y < COL and rooms[x][y] == 2147483647:
                rooms[x][y] = distance + 1
                room_q.append((x, y))
                visited.add((x, y))
                
        for i in range(ROW):
            for j in range(COL):
                if rooms[i][j] == 0:
                    room_q.append((i, j))
                    visited.add((i, j))
        distance = 0
        dirs = [(0,1), (1,0), (0, -1), (-1, 0)]
        while(room_q):
            cur_len =len(room_q)
            for i in range(cur_len):
                x, y = room_q.popleft()
                for one_dir in dirs:
                    x_new = x + one_dir[0]
                    y_new = y + one_dir[1]
                    add_room(x_new, y_new)                        
            distance += 1
```
